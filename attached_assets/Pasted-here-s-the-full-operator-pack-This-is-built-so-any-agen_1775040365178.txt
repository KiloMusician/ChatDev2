4. PowerShell bootstrap command set

This is the operator-friendly bootstrap for PowerShell 7 / Windows PowerShell.

# === NuSyQ Ecosystem Bootstrap ===
$env:NUSYQ_HOME               = "C:\prime_anchor"
$env:NUSYQ_HUB                = "C:\prime_anchor\NuSyQ-Hub"
$env:DEV_MENTOR               = "C:\Users\keath\Dev-Mentor"
$env:TERMINAL_DEPTHS_ROOT     = "C:\Users\keath\Dev-Mentor"
$env:SIMULATEDVERSE_ROOT      = "C:\prime_anchor\SimulatedVerse\SimulatedVerse"
$env:NUSYQ_ULTIMATE_ROOT      = "C:\prime_anchor\NuSyQ-Ultimate"
$env:CONCEPT_SAMURAI_ROOT     = "C:\prime_anchor\CONCEPT-SAMURAI"

$paths = @(
  $env:NUSYQ_HUB,
  $env:DEV_MENTOR,
  $env:SIMULATEDVERSE_ROOT,
  $env:NUSYQ_ULTIMATE_ROOT,
  $env:CONCEPT_SAMURAI_ROOT
) | Where-Object { $_ -and (Test-Path $_) }

$env:PYTHONPATH = (($paths -join ";") + ";" + $env:PYTHONPATH).Trim(";")

Import-Module "$env:NUSYQ_HUB\scripts\TerminalDepths.psm1" -Force

function cdhub    { Set-Location $env:NUSYQ_HUB }
function cdtd     { Set-Location $env:DEV_MENTOR }
function cdverse  { Set-Location $env:SIMULATEDVERSE_ROOT }
function cdult    { Set-Location $env:NUSYQ_ULTIMATE_ROOT }
function cdsam    { Set-Location $env:CONCEPT_SAMURAI_ROOT }

function ecoscan {
  Write-Host "=== Repo Scan ==="
  $paths | ForEach-Object { Write-Host $_ }
  Write-Host "`n=== PYTHONPATH ==="
  Write-Host $env:PYTHONPATH
  Write-Host "`n=== TD Launcher Test ==="
  Get-Command td -ErrorAction SilentlyContinue
}

function tdboot {
  td -Mode daemon -Surface web -Port 7337
}

function tdcli {
  td -Mode cli -Surface terminal
}

function tdagent {
  param([string]$Agent = "Serena")
  td -Mode agent -Agent $Agent -Surface terminal
}

function tdhealth {
  curl http://localhost:7337/api/health
  curl http://localhost:7337/api/manifest
  curl http://localhost:7337/api/system/autoboot
  curl http://localhost:7337/api/nusyq/status
  curl http://localhost:7337/api/serena/status
}

This directly builds on the existing td function/module design already present in TerminalDepths.psm1.
