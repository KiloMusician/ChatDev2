PowerShell steward script

Save as sync-steward.ps1 somewhere shared, ideally in your ecosystem utilities folder.

param(
    [string[]]$RepoPaths = @(),
    [switch]$Full,
    [switch]$Hourly,
    [switch]$DryRun,
    [string]$LogDir = "$PSScriptRoot\logs",
    [string]$BranchFallback = "main"
)

$ErrorActionPreference = "Stop"

function Write-Section {
    param([string]$Text)
    Write-Host ""
    Write-Host ("=" * 72)
    Write-Host $Text
    Write-Host ("=" * 72)
}

function Ensure-Dir {
    param([string]$Path)
    if (-not (Test-Path $Path)) {
        New-Item -ItemType Directory -Path $Path -Force | Out-Null
    }
}

function Test-GitRepo {
    param([string]$Path)
    return (Test-Path (Join-Path $Path ".git"))
}

function Invoke-Git {
    param(
        [string]$RepoPath,
        [string[]]$Args,
        [switch]$AllowFailure
    )

    Push-Location $RepoPath
    try {
        $output = & git @Args 2>&1
        $code = $LASTEXITCODE
        if ($code -ne 0 -and -not $AllowFailure) {
            throw "git $($Args -join ' ') failed in $RepoPath`n$output"
        }
        return [PSCustomObject]@{
            Code = $code
            Output = ($output -join "`n")
        }
    }
    finally {
        Pop-Location
    }
}

function Get-DefaultRepoPaths {
    $candidates = @(
        $env:DEV_MENTOR,
        $env:NUSYQ_HUB,
        $env:SIMULATEDVERSE_ROOT,
        $env:NUSYQ_ULTIMATE_ROOT,
        $env:CONCEPT_SAMURAI_ROOT
    ) | Where-Object { $_ -and (Test-Path $_) }

    return $candidates
}

function Get-BranchName {
    param([string]$RepoPath)
    $r = Invoke-Git -RepoPath $RepoPath -Args @("branch", "--show-current")
    $name = $r.Output.Trim()
    if ([string]::IsNullOrWhiteSpace($name)) { return $BranchFallback }
    return $name
}

function Get-StatusShort {
    param([string]$RepoPath)
    $r = Invoke-Git -RepoPath $RepoPath -Args @("status", "-sb")
    return $r.Output.Trim()
}

function Get-RemoteState {
    param([string]$RepoPath, [string]$Branch)

    $upstreamRef = "origin/$Branch"

    $verify = Invoke-Git -RepoPath $RepoPath -Args @("rev-parse", "--verify", $upstreamRef) -AllowFailure
    if ($verify.Code -ne 0) {
        return [PSCustomObject]@{
            UpstreamExists = $false
            Ahead = 0
            Behind = 0
            Diverged = $false
            Summary = "no upstream"
        }
    }

    $counts = Invoke-Git -RepoPath $RepoPath -Args @("rev-list", "--left-right", "--count", "HEAD...$upstreamRef")
    $parts = $counts.Output.Trim().Split(" ", [System.StringSplitOptions]::RemoveEmptyEntries)
    $ahead = [int]$parts[0]
    $behind = [int]$parts[1]

    return [PSCustomObject]@{
        UpstreamExists = $true
        Ahead = $ahead
        Behind = $behind
        Diverged = ($ahead -gt 0 -and $behind -gt 0)
        Summary = if ($ahead -eq 0 -and $behind -eq 0) {
            "up to date"
        } elseif ($ahead -gt 0 -and $behind -eq 0) {
            "ahead $ahead"
        } elseif ($ahead -eq 0 -and $behind -gt 0) {
            "behind $behind"
        } else {
            "diverged: ahead $ahead / behind $behind"
        }
    }
}

function Test-WorkingTreeClean {
    param([string]$RepoPath)
    $r = Invoke-Git -RepoPath $RepoPath -Args @("status", "--porcelain")
    return [string]::IsNullOrWhiteSpace($r.Output)
}

function New-AutoCommitMessage {
    param([string]$RepoName)
    $stamp = Get-Date -Format "yyyy-MM-dd HH:mm"
    return "chore(sync): steward checkpoint for $RepoName @ $stamp"
}

function Run-Validation {
    param([string]$RepoPath)

    $results = New-Object System.Collections.Generic.List[string]

    if (Test-Path (Join-Path $RepoPath "scripts\validate_all.py")) {
        Push-Location $RepoPath
        try {
            $out = & python "scripts\validate_all.py" 2>&1
            $results.Add("validate_all.py: executed")
            $results.Add(($out -join "`n"))
        }
        catch {
            $results.Add("validate_all.py: failed")
            $results.Add($_.Exception.Message)
        }
        finally {
            Pop-Location
        }
    }
    elseif (Test-Path (Join-Path $RepoPath "package.json")) {
        Push-Location $RepoPath
        try {
            $out = & npm test -- --runInBand 2>&1
            $results.Add("npm test: executed")
            $results.Add(($out -join "`n"))
        }
        catch {
            $results.Add("npm test: failed")
            $results.Add($_.Exception.Message)
        }
        finally {
            Pop-Location
        }
    }
    else {
        $results.Add("no standard validation script detected")
    }

    return ($results -join "`n")
}

function Sync-OneRepo {
    param(
        [string]$RepoPath,
        [switch]$Full,
        [switch]$DryRun
    )

    $repoName = Split-Path $RepoPath -Leaf
    $branch = Get-BranchName -RepoPath $RepoPath

    Write-Section "SYNC :: $repoName :: $branch"

    $actions = New-Object System.Collections.Generic.List[string]
    $filesTouched = @()
    $validation = ""
    $blockers = New-Object System.Collections.Generic.List[string]

    $statusBefore = Get-StatusShort -RepoPath $RepoPath
    Write-Host $statusBefore

    $remotes = Invoke-Git -RepoPath $RepoPath -Args @("remote", "-v")
    Write-Host $remotes.Output

    if (-not $DryRun) {
        $null = Invoke-Git -RepoPath $RepoPath -Args @("fetch", "--all", "--prune")
        $actions.Add("fetched")
    } else {
        $actions.Add("dry-run: skipped fetch")
    }

    $remoteState = Get-RemoteState -RepoPath $RepoPath -Branch $branch
    Write-Host "Remote state: $($remoteState.Summary)"

    $isClean = Test-WorkingTreeClean -RepoPath $RepoPath

    if ($remoteState.Behind -gt 0 -and -not $remoteState.Diverged) {
        if ($isClean -and -not $DryRun) {
            $null = Invoke-Git -RepoPath $RepoPath -Args @("pull", "--rebase", "origin", $branch)
            $actions.Add("rebased from origin/$branch")
        } elseif (-not $isClean -and -not $DryRun) {
            $null = Invoke-Git -RepoPath $RepoPath -Args @("stash", "push", "-u", "-m", "sync-steward auto-stash")
            $actions.Add("stashed local work")
            $null = Invoke-Git -RepoPath $RepoPath -Args @("pull", "--rebase", "origin", $branch)
            $actions.Add("rebased from origin/$branch")
            $stashPop = Invoke-Git -RepoPath $RepoPath -Args @("stash", "pop") -AllowFailure
            if ($stashPop.Code -eq 0) {
                $actions.Add("restored stashed work")
            } else {
                $blockers.Add("stash pop needs manual conflict resolution")
            }
        } else {
            $actions.Add("dry-run: skipped pull --rebase")
        }
    }
    elseif ($remoteState.Diverged) {
        $blockers.Add("branch diverged; manual review recommended before automated push")
    }

    $statusAfterReconcile = Get-StatusShort -RepoPath $RepoPath
    Write-Host $statusAfterReconcile

    if ($Full) {
        $validation = Run-Validation -RepoPath $RepoPath
        $actions.Add("validation attempted")
    } else {
        $validation = "hourly mode: validation deferred"
    }

    $porcelain = Invoke-Git -RepoPath $RepoPath -Args @("status", "--porcelain")
    $changedLines = @()
    if (-not [string]::IsNullOrWhiteSpace($porcelain.Output)) {
        $changedLines = $porcelain.Output -split "`n" | Where-Object { $_.Trim() }
        $filesTouched = $changedLines | ForEach-Object {
            ($_ -replace '^\s*[MADRCU\?]+\s+', '').Trim()
        }
    }

    if ($Full -and $filesTouched.Count -gt 0 -and -not $DryRun) {
        $null = Invoke-Git -RepoPath $RepoPath -Args @("add", "-A")
        $actions.Add("staged")
        $commitMessage = New-AutoCommitMessage -RepoName $repoName
        $commit = Invoke-Git -RepoPath $RepoPath -Args @("commit", "-m", $commitMessage) -AllowFailure
        if ($commit.Code -eq 0) {
            $actions.Add("committed")
        } else {
            if ($commit.Output -match "nothing to commit") {
                $actions.Add("no commit created")
            } else {
                $blockers.Add("commit failed: $($commit.Output)")
            }
        }
    }

    $postCommitState = Get-RemoteState -RepoPath $RepoPath -Branch $branch
    if ($Full -and $postCommitState.Ahead -gt 0 -and -not $postCommitState.Diverged -and -not $DryRun) {
        $push = Invoke-Git -RepoPath $RepoPath -Args @("push", "origin", $branch) -AllowFailure
        if ($push.Code -eq 0) {
            $actions.Add("pushed")
        } else {
            $blockers.Add("push failed: $($push.Output)")
        }
    }

    $finalState = Get-RemoteState -RepoPath $RepoPath -Branch $branch

    return [PSCustomObject]@{
        Repo = $repoName
        Path = $RepoPath
        Branch = $branch
        RemoteState = $finalState.Summary
        Actions = $actions
        FilesTouched = $filesTouched
        Validation = $validation
        Blockers = $blockers
    }
}

Ensure-Dir -Path $LogDir

if (-not $RepoPaths -or $RepoPaths.Count -eq 0) {
    $RepoPaths = Get-DefaultRepoPaths
}

$RepoPaths = $RepoPaths | Where-Object { $_ -and (Test-Path $_) -and (Test-GitRepo $_) } | Select-Object -Unique

if ($RepoPaths.Count -eq 0) {
    throw "No valid git repositories found."
}

$modeLabel = if ($Full) { "full" } elseif ($Hourly) { "hourly" } else { "standard" }
$now = Get-Date
$stamp = $now.ToString("yyyyMMdd_HHmmss")
$logFile = Join-Path $LogDir "sync_steward_$modeLabel`_$stamp.log"

$reports = foreach ($repo in $RepoPaths) {
    Sync-OneRepo -RepoPath $repo -Full:$Full -DryRun:$DryRun
}

$reportText = foreach ($r in $reports) {
@"
repo: $($r.Repo)
branch: $($r.Branch)
remote_sync_state: $($r.RemoteState)
actions: $([string]::Join(", ", $r.Actions))
files_touched: $([string]::Join(", ", $r.FilesTouched))
validation:
$($r.Validation)
remaining_blockers: $([string]::Join(" | ", $r.Blockers))
next_sync: $(if ($Hourly) { "top of next hour" } elseif ($Full) { "3 hours (or sooner if active)" } else { "1 hour lightweight / 3 hour full" })

"@
}

$reportText | Set-Content -Path $logFile -Encoding UTF8

Write-Section "SYNC STEWARD SUMMARY"
$reportText
Write-Host "Log written to: $logFile"
How to run it

For a lightweight hourly check:

powershell -ExecutionPolicy Bypass -File .\sync-steward.ps1 -Hourly

For a full reconcile cycle:

powershell -ExecutionPolicy Bypass -File .\sync-steward.ps1 -Full

For a safe preview:

powershell -ExecutionPolicy Bypass -File .\sync-steward.ps1 -Full -DryRun
Scheduled task setup

This creates two Windows scheduled tasks: one hourly lightweight pass and one every 3 hours full pass.

Save this as register-sync-steward-tasks.ps1:

param(
    [string]$ScriptPath = "C:\prime_anchor\NuSyQ-Hub\scripts\sync-steward.ps1"
)

$hourlyAction = New-ScheduledTaskAction `
    -Execute "powershell.exe" `
    -Argument "-ExecutionPolicy Bypass -File `"$ScriptPath`" -Hourly"

$hourlyTrigger = New-ScheduledTaskTrigger -Once -At (Get-Date).Date `
| ForEach-Object {
    $_.RepetitionInterval = "PT1H"
    $_.RepetitionDuration = "P1D"
    $_
}

$fullAction = New-ScheduledTaskAction `
    -Execute "powershell.exe" `
    -Argument "-ExecutionPolicy Bypass -File `"$ScriptPath`" -Full"

$fullTrigger = New-ScheduledTaskTrigger -Once -At (Get-Date).Date `
| ForEach-Object {
    $_.RepetitionInterval = "PT3H"
    $_.RepetitionDuration = "P1D"
    $_
}

Register-ScheduledTask -TaskName "NuSyQ-Sync-Steward-Hourly" -Action $hourlyAction -Trigger $hourlyTrigger -Force
Register-ScheduledTask -TaskName "NuSyQ-Sync-Steward-Full" -Action $fullAction -Trigger $fullTrigger -Force

Run it with:

powershell -ExecutionPolicy Bypass -File .\register-sync-steward-tasks.ps1
TerminalDepths wrapper idea

If you want this reachable from your cockpit, expose it as a TerminalDepths command, for example:

sync steward hourly
sync steward full
sync steward dryrun

The wrapper can just shell out to the PowerShell script.

Example command mapping logic:

# pseudo-handler
def _cmd_sync(args):
    if args == ["steward", "hourly"]:
        return run_ps("sync-steward.ps1 -Hourly")
    if args == ["steward", "full"]:
        return run_ps("sync-steward.ps1 -Full")
    if args == ["steward", "dryrun"]:
        return run_ps("sync-steward.ps1 -Full -DryRun")
Good operational pattern

Use this rhythm:

At the top of each hour:

fetch
inspect ahead/behind
inspect local status
log blockers

Every 3 hours:

full reconcile
validate
stage
commit
push
broadcast

If branches are conflict-heavy:

keep hourly checks
reduce full auto-persist to every 6 hours
surface blockers immediately instead of pretending the branch is healthy
Important guardrails

Do not let the script:

force-push
auto-resolve large semantic conflicts
commit giant unrelated changes
hide failed validation

Do let it:

fetch often
surface divergence early
stash/rebase safely when appropriate
checkpoint useful work
keep parallel agents informed
