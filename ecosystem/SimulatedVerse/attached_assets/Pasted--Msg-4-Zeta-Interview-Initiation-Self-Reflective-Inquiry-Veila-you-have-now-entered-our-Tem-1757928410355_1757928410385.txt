```python
def weighted_median_hub(X_list, w, C_star, k_star,
                        alpha=1.0, beta=1.0, gamma=0.2, max_sweeps=5):
    # X_list: list of bitsets (as 0/1 arrays length m)
    # w: weights (sum to 1)
    m = len(X_list[0])
    # 1) weighted bitwise median
    p = [sum(w[i]*X_list[i][j] for i in range(len(X_list))) for j in range(m)]
    Y = [1 if p[j] >= 0.5 else 0 for j in range(m)]
    # 2) local improvement
    def cost(Yvec):
        # Hamming term
        h = sum(w[i]*sum(X_list[i][j]^Yvec[j] for j in range(m))
                for i in range(len(X_list)))
        # correlation (1..6 on ring); implement C(Y) here
        C_Y = correlation_vector(Yvec)   # length 6
        corr = sum(abs(C_Y[d]-C_star[d]) for d in range(6))
        # size penalty
        size = sum(Yvec)
        return alpha*h + beta*corr + gamma*abs(size - k_star)
    base = cost(Y)
    for _ in range(max_sweeps):
        improved = False
        for j in range(m):
            Y[j] ^= 1  # flip bit
            c = cost(Y)
            if c + 1e-9 < base:
                base = c; improved = True
            else:
                Y[j] ^= 1  # revert
        if not improved:
            break
    return Y, base
```
