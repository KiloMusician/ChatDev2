## 🤖 Automated Cleanup Script

**Create**: `scripts/emergency_cleanup.py`

```python
#!/usr/bin/env python3
"""
Emergency workspace cleanup - removes bloat with safety checks
Run this when workspace exceeds size limits
"""

import os
import sys
from pathlib import Path
from datetime import datetime
import shutil

# Config
DRY_RUN = True  # Set False to actually delete
BACKUP_DIR = Path.home() / "Backups"
WORKSPACE_ROOT = Path(__file__).parent.parent

def get_file_size_gb(path):
    """Get file size in GB"""
    return os.path.getsize(path) / (1024**3)

def cleanup_maze_summaries():
    """Keep only 3 most recent maze_summary files"""
    logs_dir = WORKSPACE_ROOT / "logs"
    if not logs_dir.exists():
        return

    files = sorted(
        logs_dir.glob("maze_summary_*.json"),
        key=lambda p: p.stat().st_mtime,
        reverse=True
    )

    to_delete = files[3:]  # Keep first 3, delete rest

    print(f"\\n=== Maze Summary Cleanup ===")
    print(f"Total files: {len(files)}")
    print(f"Keeping: {len(files[:3])}")
    print(f"Deleting: {len(to_delete)}")

    total_size = sum(get_file_size_gb(f) for f in to_delete)
    print(f"Space to recover: {total_size:.2f} GB")

    if DRY_RUN:
        print("\\n[DRY RUN] Would delete:")
        for f in to_delete:
            print(f"  - {f.name} ({get_file_size_gb(f):.2f} GB)")
    else:
        print("\\nDeleting files...")
        for f in to_delete:
            print(f"  Deleting {f.name}...")
            f.unlink()
        print(f"✅ Deleted {len(to_delete)} files, recovered {total_size:.2f} GB")

def cleanup_large_artifacts():
    """Remove large generated artifacts"""
    artifacts = [
        WORKSPACE_ROOT / "docs/dependency-analysis/dependency-analysis.json",
        WORKSPACE_ROOT / "function_registry_data.json",
    ]

    print(f"\\n=== Large Artifacts Cleanup ===")
    total_size = 0

    for path in artifacts:
        if path.exists():
            size = get_file_size_gb(path)
            total_size += size
            print(f"  - {path.name} ({size:.2f} GB)")

            if not DRY_RUN:
                path.unlink()
                print(f"    ✅ Deleted")

    print(f"\\nSpace to recover: {total_size:.2f} GB")

def create_backup():
    """Create backup before cleanup"""
    if DRY_RUN:
        print("\\n[DRY RUN] Would create backup in ~/Backups/")
        return

    BACKUP_DIR.mkdir(exist_ok=True)
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    backup_file = BACKUP_DIR / f"workspace_backup_{timestamp}.tar.gz"

    print(f"\\nCreating backup: {backup_file}")
    # Backup implementation here
    print("✅ Backup created")

def main():
    global DRY_RUN

    if "--execute" in sys.argv:
        DRY_RUN = False
        print("🚨 EXECUTING CLEANUP (NOT A DRY RUN)")
        create_backup()
    else:
        print("🔍 DRY RUN MODE (use --execute to actually delete)")

    cleanup_maze_summaries()
    cleanup_large_artifacts()

    print("\\n" + "="*60)
    if DRY_RUN:
        print("To execute cleanup: python scripts/emergency_cleanup.py --execute")
    else:
        print("✅ Cleanup complete!")

if __name__ == "__main__":
    main()
```
