"""
KILO-FOOLISH Ollama Integration Hub
Local AI intelligence for privacy-first operations
"""

import json
import logging
import sys
import argparse
from pathlib import Path
from datetime import datetime
from typing import Dict, List, Optional, Any

# Add src to path for imports
sys.path.insert(0, str(Path(__file__).parent.parent))

try:
    from config.secrets import get_config
except ImportError:
    print("❌ Cannot import config - ensure secrets.py exists")
    sys.exit(1)


class KILOOllamaHub:
    """KILO-FOOLISH Ollama Intelligence Hub"""

    def __init__(self):
        # Setup logging
        logging.basicConfig(level=logging.INFO)
        self.logger = logging.getLogger(__name__)

        # Get configuration
        try:
            self.config = get_config()
            self.client = self.config.get_ollama_client()
            self.is_connected = True
            self.logger.info("Ollama client initialized successfully")
        except Exception as e:
            self.logger.error(f"Failed to initialize Ollama: {e}")
            self.is_connected = False
            self.client = None

        # Model preferences
        self.model_preferences = {
            "code_analysis": "codellama:7b",
            "documentation": "llama2:7b",
            "reasoning": "mistral:7b",
            "default": "llama2:7b"
        }

        # Available models cache
        self.available_models = {}
        if self.is_connected:
            self.discover_models()

    def discover_models(self):
        """Discover available models"""
        try:
            models_response = self.client.list()
            self.available_models = {}

            for model_info in models_response.get('models', []):
                self.available_models[model_info['name']] = {
                    'size': model_info.get('size', 0),
                    'modified_at': model_info.get('modified_at', ''),
                    'digest': model_info.get('digest', '')
                }

            self.logger.info(f"Discovered {len(self.available_models)} models")

        except Exception as e:
            self.logger.error(f"Model discovery failed: {e}")

    def select_model(self, task_type: str = "default") -> str:
        """Select appropriate model for task"""
        preferred = self.model_preferences.get(task_type, "llama2:7b")

        # Check if preferred model is available
        if preferred in self.available_models:
            return preferred

        # Fallback to first available model
        if self.available_models:
            fallback = list(self.available_models.keys())[0]
            self.logger.info(f"Using fallback model: {fallback}")
            return fallback

        # Default fallback
        return "llama2:7b"

    def chat(self, message: str, model: str = None) -> dict:
        """Chat with Ollama"""
        if not self.is_connected:
            return {"error": "Ollama not connected"}

        if not model:
            model = self.select_model("default")

        try:
            response = self.client.chat(
                model=model,
                messages=[{"role": "user", "content": message}],
                options={'temperature': 0.7}
            )

            return {
                "success": True,
                "message": response['message']['content'],
                "model": model,
                "timestamp": datetime.now().isoformat()
            }

        except Exception as e:
            self.logger.error(f"Chat failed: {e}")
            return {"error": str(e)}

    def analyze_code(self, file_path: str) -> dict:
        """Analyze code file"""
        if not self.is_connected:
            return {"error": "Ollama not connected"}

        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                code_content = f.read()
        except Exception as e:
            return {"error": f"Cannot read file: {e}"}

        analysis_prompt = f'''
        Analyze this code and provide insights:

        File: {file_path}

        Code:
        ```python
        {code_content}
        ```

        Please provide:
        1. Code quality assessment
        2. Potential improvements
        3. Security considerations
        4. Best practices recommendations
        '''

        model = self.select_model("code_analysis")
        return self.chat(analysis_prompt, model)

    def generate_docs(self, code_snippet: str) -> dict:
        """Generate documentation"""
        if not self.is_connected:
            return {"error": "Ollama not connected"}

        doc_prompt = f'''
        Generate comprehensive documentation for this code:

        ```python
        {code_snippet}
        ```

        Include:
        - Purpose and functionality
        - Parameters and return values
        - Usage examples
        - Important notes
        '''

        model = self.select_model("documentation")
        return self.chat(doc_prompt, model)

    def pull_model(self, model_name: str) -> dict:
        """Pull a new model"""
        if not self.is_connected:
            return {"error": "Ollama not connected"}

        try:
            self.logger.info(f"Pulling model: {model_name}")
            self.client.pull(model_name)

            # Refresh models
            self.discover_models()

            return {"success": True, "message": f"Model {model_name} pulled successfully"}

        except Exception as e:
            return {"error": f"Failed to pull model: {e}"}

    def get_status(self) -> dict:
        """Get hub status"""
        return {
            "connection_status": "connected" if self.is_connected else "disconnected",
            "available_models": len(self.available_models),
            "model_list": list(self.available_models.keys()),
            "model_preferences": self.model_preferences,
            "timestamp": datetime.now().isoformat()
        }


def main():
    parser = argparse.ArgumentParser(description="KILO-FOOLISH Ollama Hub")
    parser.add_argument("--chat", help="Chat message")
    parser.add_argument("--model", help="Specific model to use")
    parser.add_argument("--analyze", help="Analyze code file")
    parser.add_argument("--generate-docs", help="Generate documentation for code")
    parser.add_argument("--pull", help="Pull model")
    parser.add_argument("--status", action="store_true", help="Show status")
    parser.add_argument("--models", action="store_true", help="List models")

    args = parser.parse_args()

    hub = KILOOllamaHub()

    if args.status:
        status = hub.get_status()
        print(json.dumps(status, indent=2))

    elif args.models:
        if hub.available_models:
            print("📦 Available Models:")
            for model, info in hub.available_models.items():
                size_mb = info['size'] / (1024 * 1024) if info['size'] else 0
                print(f"  • {model} ({size_mb:.1f} MB)")
        else:
            print("No models available")

    elif args.chat:
        result = hub.chat(args.chat, args.model)
        if result.get("success"):
            print(f"🤖 [{result['model']}]: {result['message']}")
        else:
            print(f"❌ Error: {result.get('error', 'Unknown error')}")

    elif args.analyze:
        result = hub.analyze_code(args.analyze)
        if result.get("success"):
            print(f"🔍 Code Analysis [{result['model']}]:")
            print("=" * 50)
            print(result['message'])
        else:
            print(f"❌ Analysis failed: {result.get('error', 'Unknown error')}")

    elif args.generate_docs:
        result = hub.generate_docs(args.generate_docs)
        if result.get("success"):
            print(f"📝 Generated Documentation [{result['model']}]:")
            print("=" * 50)
            print(result['message'])
        else:
            print(f"❌ Documentation generation failed: {result.get('error', 'Unknown error')}")

    elif args.pull:
        result = hub.pull_model(args.pull)
        if result.get("success"):
            print(f"✅ {result['message']}")
        else:
            print(f"❌ {result.get('error', 'Unknown error')}")

    else:
        print("🤖 KILO-FOOLISH Ollama Hub")
        print("Use --help for available commands")


if __name__ == "__main__":
    main()

function Write-OllamaLog {
    param([string]`$Message, [string]`$Level = "INFO")

    `$colors = @{
        "ERROR" = "Red"; "WARNING" = "Yellow"; "SUCCESS" = "Green"
        "INFO" = "Cyan"; "OLLAMA" = "Magenta"; "CHAT" = "White"
    }

    `$timestamp = Get-Date -Format "HH:mm:ss"
    Write-Host "[`$timestamp] [OLLAMA] `$Message" -ForegroundColor `$colors[`$Level]
}

function Test-OllamaConnection {
    try {
        `$testResult = python ".\src\core\OllamaHub.py" --status 2>`$null
        return `$LASTEXITCODE -eq 0
    }
    catch {
        return `$false
    }
}

function Start-OllamaChat {
    param([string]`$Message, [string]`$Model = "")

    Write-OllamaLog "🤖 Starting Ollama chat..." "OLLAMA"

    if (!(Test-OllamaConnection)) {
        Write-OllamaLog "❌ Ollama not connected. Make sure Ollama is running:" "ERROR"
        Write-Host "   ollama serve" -ForegroundColor Yellow
        return
    }

    try {
        `$pythonArgs = @("--chat", `$Message)
        if (`$Model) { `$pythonArgs += @("--model", `$Model) }

        `$result = python ".\src\core\OllamaHub.py" @pythonArgs 2>&1

        if (`$LASTEXITCODE -eq 0) {
            Write-Host `$result -ForegroundColor White
        } else {
            Write-OllamaLog "❌ Chat failed: `$result" "ERROR"
        }
    }
    catch {
        Write-OllamaLog "❌ Chat error: `$_" "ERROR"
    }
}

            Write-Host "=" * 40 -ForegroundColor Magenta
