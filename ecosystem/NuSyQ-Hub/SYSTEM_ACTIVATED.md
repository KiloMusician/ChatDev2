from src.agents.autonomous_development_agent import AutonomousDevelopmentAgent
import asyncio

async def develop():
    agent = AutonomousDevelopmentAgent()
    await agent.initialize_systems()

    # Multi-agent game development

        "puzzle game with level editor",
        complexity="medium"
